
// 사진 올리기
create table photo_post(
    idx     int     auto_increment primary key,
    user_email varchar(100) null,
    pyeong  varchar(100) not null,
    type    varchar(100) not null,
    style   varchar(100) not null,
    file    longblob not null,
    place   varchar(100) not null,
    memo    varchar(500) not null,
    write_date timestamp default current_timestamp
);

// 동영상 올리기
create table video_post(
    idx     int     auto_increment primary key,
    user_email  varchar(100) null,
    video_pyeong  varchar(100) not null,
    video_type    varchar(100) not null,
    video_style   varchar(100) not null,
    video_file    longblob not null,
    video_place   varchar(100) not null,
    video_memo    varchar(500) not null,
    write_date timestamp default current_timestamp
);


// 집들이 올리기
create table house_post(
    idx     int     auto_increment primary key,
    file    longblob not null,
    house_title varchar(81) not null,
    house_content varchar(2000) not null,
    write_date timestamp default current_timestamp
);
